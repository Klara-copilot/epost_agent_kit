```swift
// Modal Presentation (in presenting view controller)
func presentModal() {
    let modalVC = ModalViewController()
    present(modalVC, animated: true) {
        // Set modal accessibility
        modalVC.view.accessibilityViewIsModal = true
        
        // Focus first element
        UIAccessibility.post(
            notification: .screenChanged,
            argument: modalVC.firstElement
        )
        
        // Announce modal
        UIAccessibility.post(
            notification: .announcement,
            argument: "Modal dialog opened"
        )
    }
}

// Modal Dismissal (in modal view controller)
func dismissModal() {
    dismiss(animated: true) {
        // Return focus to triggering element
        UIAccessibility.post(
            notification: .screenChanged,
            argument: self.presentingViewController?.triggeringButton
        )
    }
}
```

**Critical:** Ensure dismiss button is always accessible to prevent focus traps.
